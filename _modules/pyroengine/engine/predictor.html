import torch



[docs]class PyronearPredictor:
    """This class use the last pyronear model and run our smoke detection model on it
    Examples:
        >>> pyronearPredictor = PyronearPredictor()
        >>> im = Image.open("image.jpg")
        >>> res = pyronearPredictor.predict(im)
    """

    def __init__(self, model_weights: str = None, conf: float = 0.25):
        """Init predictor"""
        # Model definition
        if model_weights is None:
            model_weights = "https://github.com//pyronear//pyro-vision//releases//download//v0.1.2//yolov5s_v001.onnx"
        self.model = torch.hub.load(
            "ultralytics/yolov5", "custom", path=model_weights, force_reload=True
        )  # local model
        self.model.conf = conf

    def predict(self, im):
        """Run prediction"""
        pred = self.model(im)

        return pred.xyxy[0].numpy()
